#include<stdio.h>
int main(){
    int a = 23;
    printf("%d\n",a);
    
    int b = 65;
    printf("%d\n",b);
    
    int c = -23;
    printf("%d\n",c);
    
    float d = 1.56;
    printf("%f\n",d);
    
    float e = 0.8;
    printf("%f\n",e);
    
    float f = -2.752;
    printf("%f\n",f);
    
    char g = 't';
    printf("%c\n",g);
    
    int h = 123456;
    printf("%d\n",h);
    
    char i = 'o';
    printf("%c\n",i);
    
    return 0;
}
